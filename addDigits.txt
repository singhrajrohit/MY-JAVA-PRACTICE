import java.util.Scanner;

public class addDigits{
    public static void main(String[] args) {
        
        Scanner sc=new Scanner(System.in);
        System.out.println("Enter Number");
        int n=sc.nextInt();
        int sum=0;
        
        while(n>0){
            sum+=n%10;
            n=n/10;
        }

        System.out.println("sum of digits =" +sum);
    }
}
